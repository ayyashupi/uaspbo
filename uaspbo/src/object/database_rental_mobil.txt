drop database rental_pbo;
create database rental_pbo;
use rental_pbo;

create table mobil(
id_mobil int primary key auto_increment,
nopol varchar(20),
model varchar(20),
jml_penumpang int,
transmisi varchar(20),
BBM varchar(20),
harga_sewa int,
status varchar(20)
);


create table supir(
id_supir int primary key auto_increment,
nama varchar(20),
no_telp varchar(20),
biaya int,
jml_cus int,
status varchar(20)
);

create table penyewa(
id_penyewa int primary key auto_increment,
nama varchar(20),
username varchar(20),
password varchar(20),
gender varchar(20),
no_telp varchar(20),
alamat varchar(50),
no_ktp varchar(20),
saldo int
);

create table transaksi(
id_transaksi int primary key auto_increment,
id_penyewa int,
kd_mobil int,
id_supir int,
tgl_pinjam date,
tgl_kembali date,
diskon int,
total int,
foreign key (id_penyewa) references penyewa(id_penyewa),
foreign key (kd_mobil) references mobil(id_mobil),
foreign key (id_supir) references supir(id_supir)
);

delimiter %%
create or replace trigger addJmlCusDriver
after insert
on transaksi FOR EACH ROW    
begin

update supir
set jml_cus = jml_cus+1
where id_supir = new.id_supir;

end %%
delimiter ;



delimiter %%
create or replace trigger editJmlCusDriver
after update
on transaksi FOR EACH ROW    
begin

update supir
set jml_cus = jml_cus+1
where id_supir = new.id_supir;

update supir
set jml_cus = jml_cus-1
where id_supir = old.id_supir;

end %%
delimiter ;



delimiter %%
create or replace trigger deleteJmlCusDriver
before delete
on transaksi FOR EACH ROW    
begin

update supir
set jml_cus = jml_cus-1
where id_supir = old.id_supir;

end %%
delimiter ;

delimiter %%
create or replace trigger tambahSaldo
before delete
on transaksi FOR EACH ROW    
begin

update penyewa
set saldo = saldo+old.total
where id_penyewa = old.id_penyewa;

end %%
delimiter ;
